function role_access(u){
    if (u == "admin"){
        return "Admin access granted."
    }
    else if (u == "hr"){
        return "Hr access granted."
    }
    else if (u == "employee"){
        return "Employee access granted."
    }
    else{
        return "Invalid access"
    }
}
